procedure main() returns integer {
var out = std_output();
var a = 10.0;
var b = 5.0;

a * b -> out;
'\n' -> out;
a - b -> out;
'\n' -> out;
a / b -> out;
'\n' -> out;
a + b -> out;
'\n' -> out;
a * -b -> out;
'\n' -> out;
a - -b -> out;
'\n' -> out;
a / -b -> out;
'\n' -> out;
a + -b -> out;
'\n' -> out;


return 0;
}
